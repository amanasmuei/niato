```typescript
import React, { useState } from "react";
import { Box, Text, useInput } from "ink";
import { TextInput } from "../components/text-input.js";

export interface ApiKeyEntryProps {
  onSubmit: (apiKey: string) => void;
  onCancel: () => void;
}

// Simple length guard: reject empty. The Anthropic-prefix check is a hint
// (most common misuse: pasting an OpenAI key here), not a hard reject —
// real validation happens at first SDK call. The user can confirm a
// non-conforming value with a second Enter so we don't lock them out
// of edge cases like a legitimate key from a future SDK rev.
const ANTHROPIC_PREFIX = "sk-ant-";

export function ApiKeyEntry({
  onSubmit,
  onCancel,
}: ApiKeyEntryProps): React.ReactElement {
  const [value, setValue] = useState("");
  const [error, setError] = useState<string | null>(null);
  const [pendingConfirm, setPendingConfirm] = useState(false);

  useInput((_input, key) => {
    if (key.escape) onCancel();
  });

  const handleSubmit = (v: string): void => {
    if (v.length === 0) {
      setError("API key is required. Enter a value or press Esc to go back.");
      setPendingConfirm(false);
      return;
    }
    if (!v.startsWith(ANTHROPIC_PREFIX) && !pendingConfirm) {
      setError(
        `Anthropic keys usually start with '${ANTHROPIC_PREFIX}'. ` +
          "Press Enter again to use this value anyway, or type a different key.",
      );
      setPendingConfirm(true);
      return;
    }
    onSubmit(v);
  };

  return (
    <Box flexDirection="column" paddingX={1}>
      <Text bold color="cyan">
        API key
      </Text>
      <Text color="gray">
        Paste your ANTHROPIC_API_KEY (saved to ~/.niato/auth.json,
        chmod 600).
      </Text>
      <Box marginTop={1}>
        <TextInput
          value={value}
          placeholder="sk-ant-..."
          onChange={(v) => {
            setValue(v);
            // Clear hint state as soon as the user keeps editing — they
            // saw the warning, no need to re-confirm a different value.
            if (error !== null) setError(null);
            if (pendingConfirm) setPendingConfirm(false);
          }}
          onSubmit={handleSubmit}
        />
      </Box>
      {error !== null ? (
        <Box marginTop={1}>
          <Text color="yellow">{error}</Text>
        </Box>
      ) : null}
      <Box marginTop={1}>
        <Text color="gray" dimColor>
          Enter to save · Esc to go back
        </Text>
      </Box>
    </Box>
  );
}
```

- [ ] **Step 3: Implement `src/cli/tui/screens/companion-wizard.tsx`**

```typescript
import React, { useState } from "react";
import { Box, Text, useInput } from "ink";
import { TextInput } from "../components/text-input.js";
import { Menu, type MenuItem } from "../components/menu.js";
import {
  saveCompanion,
  VOICE_ARCHETYPES,
  type Companion,
  type VoiceArchetype,
} from "../../companion-config.js";

export interface CompanionWizardProps {
  onComplete: (companion: Companion) => void;
  onCancel: () => void;
  companionPath?: string;
}

type Step = "name" | "userName" | "voice" | "description";

function toVoice(id: string): VoiceArchetype {
  // Menu emits string ids; the menu items below match VOICE_ARCHETYPES
  // exactly. Defensive fallback to "warm" if anything else slips in.
  if (id === "warm" || id === "direct" || id === "playful") return id;
  return "warm";
}

export function CompanionWizard({
  onComplete,
  onCancel,
  companionPath,
}: CompanionWizardProps): React.ReactElement {
  const [step, setStep] = useState<Step>("name");
  const [name, setName] = useState("");
  const [userName, setUserName] = useState("");
  const [extraDescription, setExtraDescription] = useState("");
  const [error, setError] = useState<string | null>(null);

  // Esc at step 1 cancels; at later steps the user is mid-flow and Esc
  // goes back to the previous step. Keeps the cancel surface predictable.
  useInput((_input, key) => {
    if (!key.escape) return;
    if (step === "name") onCancel();
    else if (step === "userName") setStep("name");
    else if (step === "voice") setStep("userName");
    else if (step === "description") setStep("voice");
  });

  const submitName = (v: string): void => {
    if (v.length === 0) {
      setError("Companion name is required.");
      return;
    }
    setName(v);
    setError(null);
    setStep("userName");
  };

  const submitUserName = (v: string): void => {
    setUserName(v);
    setStep("voice");
  };

  const submitVoice = (id: string): void => {
    const voice = toVoice(id);
    finalize(voice, extraDescription);
    setStep("description");
  };

  const submitDescription = (v: string): void => {
    setExtraDescription(v);
    finalize(undefined, v);
  };

  const finalize = (
    voiceArg: VoiceArchetype | undefined,
    descriptionArg: string,
  ): void => {
    // We finalize when description step submits. The voice step transitions
    // forward to description but does not finalize on its own — early exit
    // here when called from voice step.
    if (voiceArg !== undefined) return;

    const companion: Companion = {
      version: 1,
      name,
      voice: pickedVoice,
      createdAt: new Date().toISOString(),
      ...(userName.length > 0 ? { userName } : {}),
      ...(descriptionArg.length > 0
        ? { extraDescription: descriptionArg }
        : {}),
    };
    saveCompanion(companion, companionPath);
    onComplete(companion);
  };

  // The voice picked in step 3, held in state because finalize runs from
  // the description step.
  const [pickedVoice, setPickedVoice] = useState<VoiceArchetype>("warm");

  if (step === "name") {
    return (
      <Box flexDirection="column" paddingX={1}>
        <Text bold color="cyan">
          Set up your companion
        </Text>
        <Text color="gray">Step 1 of 4 — companion name (required)</Text>
        <Box marginTop={1}>
          <TextInput
            value={name}
            placeholder="e.g. Layla, Sage, Arienz..."
            onChange={(v) => {
              setName(v);
              if (error !== null) setError(null);
            }}
            onSubmit={submitName}
          />
        </Box>
        {error !== null ? (
          <Box marginTop={1}>
            <Text color="yellow">{error}</Text>
          </Box>
        ) : null}
        <Box marginTop={1}>
          <Text color="gray" dimColor>
            Enter to continue · Esc to cancel
          </Text>
        </Box>
      </Box>
    );
  }

  if (step === "userName") {
    return (
      <Box flexDirection="column" paddingX={1}>
        <Text bold color="cyan">
          Set up your companion
        </Text>
        <Text color="gray">Step 2 of 4 — your name (optional)</Text>
        <Box marginTop={1}>
          <TextInput
            value={userName}
            placeholder="leave empty to skip"
            onChange={setUserName}
            onSubmit={submitUserName}
          />
        </Box>
        <Box marginTop={1}>
          <Text color="gray" dimColor>
            Enter to continue · Esc to go back
          </Text>
        </Box>
      </Box>
    );
  }

  if (step === "voice") {
    const items: MenuItem[] = VOICE_ARCHETYPES.map((v) => ({
      id: v,
      label: v,
      detail:
        v === "warm"
          ? "warm + supportive"
          : v === "direct"
            ? "concise + pragmatic"
            : "playful + curious",
    }));
    return (
      <Box flexDirection="column" paddingX={1}>
        <Text bold color="cyan">
          Set up your companion
        </Text>
        <Text color="gray">Step 3 of 4 — voice archetype</Text>
        <Box marginTop={1}>
          <Menu
            items={items}
            onSelect={(id) => {
              const voice = toVoice(id);
              setPickedVoice(voice);
              setStep("description");
            }}
            onCancel={() => setStep("userName")}
          />
        </Box>
      </Box>
    );
  }

  // step === "description"
  return (
    <Box flexDirection="column" paddingX={1}>
      <Text bold color="cyan">
        Set up your companion
      </Text>
      <Text color="gray">
        Step 4 of 4 — extra description (optional)
      </Text>
      <Box marginTop={1}>
        <TextInput
          value={extraDescription}
          placeholder="leave empty to skip"
          onChange={setExtraDescription}
          onSubmit={submitDescription}
        />
      </Box>
      <Box marginTop={1}>
        <Text color="gray" dimColor>
          Enter to finish · Esc to go back
        </Text>
      </Box>
    </Box>
  );
}
```

```typescript
    saveAuth({ mode: "subscription" }, authPath);
    setAuth({ mode: "subscription" });
    proceedAfterAuth();
    return;
  }
  // api-key path: env-var shortcut still wins for the rare power user,
  // otherwise show the in-app key-entry screen.
  const env = process.env["ANTHROPIC_API_KEY"];
  if (typeof env === "string" && env.length > 0) {
    const next: AuthState = { mode: "api-key", apiKey: env };
    saveAuth(next, authPath);
    setAuth(next);
    proceedAfterAuth();
    return;
  }
  stack.replace("api-key-entry", {});
};

const proceedAfterAuth = (): void => {
  if (loadCompanion(companionPath) === null) {
    stack.replace("companion-wizard", {});
    return;
  }
  setCompanion(loadCompanion(companionPath));
  stack.replace("launcher", {});
};

const onApiKeySubmit = (apiKey: string): void => {
  const next: AuthState = { mode: "api-key", apiKey };
  saveAuth(next, authPath);
  setAuth(next);
  proceedAfterAuth();
};

const onApiKeyCancel = (): void => {
  stack.replace("first-run", {});
};

const onCompanionComplete = (companion: Companion): void => {
  setCompanion(companion);
  stack.replace("launcher", {});
};

const onCompanionCancel = (): void => {
  // From first-run flow: companion is required. Cancel goes back to
  // first-run so the user can re-pick auth or quit. From Settings:
  // we route through a different stack push (see Settings handler
  // below), so the companion's onCancel pops to whatever was beneath.
  stack.pop();
};
```
