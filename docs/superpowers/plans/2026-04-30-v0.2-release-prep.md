# v0.2.0 Release Prep — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship Niato v0.2.0 — the first version that can legally and technically be installed by anyone via `npm i -g niato`. Resolves the three blockers from the v1 roadmap that gate distribution at all: license, subscription-auth ToS posture, and npm publishability.

**Architecture:** Mostly configuration and decision-recording. One real behavior change: `resolveAuthMode` in `src/core/config.ts` requires explicit opt-in instead of silently landing on the ToS-uncertain OAuth subscription path. Throws a typed `NiatoAuthError` when neither path is configured. The TUI keeps working because its persisted auth flow sets the opt-in env var before invoking the factory.

**Tech Stack:** TypeScript strict · zod · vitest · Node 20+. No new dependencies.

**Roadmap reference:** `docs/superpowers/plans/2026-04-30-v1-release-roadmap.md` (Plan 1 of 4).

```typescript
import { NiatoAuthError } from "./errors.js";

// ... (EnvSchema and loadConfig unchanged)

// Resolve the auth path the Agent SDK will take. Two valid paths:
//
//   1. ANTHROPIC_API_KEY env var — developer API billing, per-token.
//   2. NIATO_AUTH=subscription — opt-in OAuth subscription path. Requires
//      a prior `claude /login`. ToS considerations apply for non-personal
//      use; see README "Note on subscription auth".
//
// Subscription opt-in wins when both are set: explicit intent beats default.
// When neither is set, we throw rather than silently landing on OAuth — that
// would push strangers onto the ToS-uncertain path without their knowledge.
export function resolveAuthMode(
  config: Config,
  env: NodeJS.ProcessEnv = process.env,
): AuthMode {
  if (env["NIATO_AUTH"] === "subscription") return "oauth_subscription";
  if (config.ANTHROPIC_API_KEY !== undefined) return "api_key";
  throw new NiatoAuthError(
    "No authentication configured.\n\n" +
      "Pick one:\n" +
      "  • Set ANTHROPIC_API_KEY (developer API path, per-token billing).\n" +
      "  • Set NIATO_AUTH=subscription to use Claude subscription auth\n" +
      "    (review ToS notes in README before non-personal use).\n",
  );
}
```

- [ ] **Step 3: Create `src/cli/tui/auth-env.ts`**

```typescript
import { loadAuth } from "./store/auth.js";

// Bridges the TUI's persisted auth choice to the orchestrator's env-var-based
// auth resolution (src/core/config.ts:resolveAuthMode). When the user picked
// the subscription path in first-run, we set NIATO_AUTH=subscription so
// the Agent SDK takes the OAuth path. Called once at TUI startup, before
// niatoFactory is invoked.
//
// Pre-existing NIATO_AUTH wins — explicit shell config beats persisted UI
// choice. api-key mode never sets the env: ANTHROPIC_API_KEY (set by the
// in-app key-entry flow in v0.3) carries that path on its own.
export function applyPersistedAuthEnv(authPath?: string): void {
  if (typeof process.env["NIATO_AUTH"] === "string") return;
  const persisted = authPath !== undefined ? loadAuth(authPath) : loadAuth();
  if (persisted === null) return;
  if (persisted.mode === "subscription") {
    process.env["NIATO_AUTH"] = "subscription";
  }
}
```

```typescript
// Pure subcommand router for bin/niato. The bin script handles process
// concerns (spawning, exit codes, version reading); this module owns the
// arg → entry mapping so it can be unit tested without spinning subprocesses.

export type DispatchResult =
  | { kind: "entry"; entry: string; forwardArgs: string[] }
  | { kind: "version" }
  | { kind: "help" }
  | { kind: "unknown"; subcommand: string };

const ENTRIES: Record<string, string> = {
  tui: "cli/tui/index.js",
  chat: "cli-chat.js",
  login: "cli-login.js",
};

export function resolveDispatch(args: readonly string[]): DispatchResult {
  const first = args[0];
  if (first === undefined) {
    return { kind: "entry", entry: ENTRIES["tui"]!, forwardArgs: [] };
  }
  if (first === "--version" || first === "-v") return { kind: "version" };
  if (first === "--help" || first === "-h") return { kind: "help" };

  const entry = ENTRIES[first];
  if (entry !== undefined) {
    return { kind: "entry", entry, forwardArgs: args.slice(1) };
  }
  return { kind: "unknown", subcommand: first };
}

export function helpText(): string {
  return [
    "niato — intent-routing agent on the Claude Agent SDK",
    "",
    "Usage:",
    "  niato                 launch the TUI (default)",
    "  niato tui             same",
    "  niato chat [--reset]  legacy multi-turn REPL",
    "  niato login           OAuth subscription auth (wraps `claude /login`)",
    "  niato --version       print version",
    "  niato --help          show this help",
    "",
    "Auth (set one in your shell):",
    "  ANTHROPIC_API_KEY=sk-ant-...    developer API path (recommended)",
    "  NIATO_AUTH=subscription       Claude subscription path (review ToS)",
  ].join("\n");
}
```

```javascript
#!/usr/bin/env node
// Niato CLI dispatcher (Node-based, no pnpm dependency).
// Routes subcommands to compiled entries under dist/. See
// src/cli/dispatch.ts for the pure routing logic.

import { spawn } from "node:child_process";
import { readFileSync } from "node:fs";
import { fileURLToPath } from "node:url";
import { dirname, join } from "node:path";
import { resolveDispatch, helpText } from "../dist/cli/dispatch.js";

const here = dirname(fileURLToPath(import.meta.url));
const pkgRoot = join(here, "..");
const distRoot = join(pkgRoot, "dist");

const args = process.argv.slice(2);
const result = resolveDispatch(args);

if (result.kind === "version") {
  const pkg = JSON.parse(readFileSync(join(pkgRoot, "package.json"), "utf8"));
  process.stdout.write(`${pkg.version}\n`);
  process.exit(0);
}

if (result.kind === "help") {
  process.stdout.write(`${helpText()}\n`);
  process.exit(0);
}

if (result.kind === "unknown") {
  process.stderr.write(`niato: unknown subcommand '${result.subcommand}'\n`);
  process.stderr.write("Run `niato --help` for usage.\n");
  process.exit(2);
}

const child = spawn(
  process.execPath,
  [join(distRoot, result.entry), ...result.forwardArgs],
  { stdio: "inherit" },
);
child.on("exit", (code, signal) => {
  if (signal !== null) process.kill(process.pid, signal);
  else process.exit(code ?? 1);
});
```
